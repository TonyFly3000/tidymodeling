raw_tibble001 <- text001 %>% 
  str_split(pattern = "\n") %>% 
  unlist() %>%
  tibble(data = .) %>% slice(6:n()) %>% filter(str_length(data) >= 2) %>%  mutate(old=data
         ,data=data %>%str_replace_all('_','')%>%str_replace_all('—','')%>%str_replace_all("'",'')%>%str_replace_all('"','')
         ) %>% separate(
    data, 
    into = c("one", "price"), 
    sep = c("¥")
    ) %>% mutate(
    id = str_extract(one, "^\\S+ ")
    ,name=gsub("^\\S+ ", "",one)
