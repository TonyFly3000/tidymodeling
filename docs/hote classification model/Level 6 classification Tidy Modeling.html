3.1 recipe

becasue the target class is not balance so using downsample

Code 
data_rec <- recipe(target_variable ~ ., data = data_train) %>%
  step_rm(rownum) %>% 
  step_downsample(target_variable) %>%
  step_dummy(all_nominal(), -all_outcomes()) %>%
  step_zv(all_numeric()) %>%
  step_normalize(all_numeric())
